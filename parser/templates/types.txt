module bindbc.raylib.types;

enum RaylibSupport {
   noLibrary,
   badLibrary,
   raylib370 = 370
}

enum RAYLIB_VERSION_MAJOR = 3;

version (RAYLIB_370) {
   enum RAYLIB_VERSION_MINOR = 7;
   enum RAYLIB_VERSION_REVISION = 0;
   enum raylibSupport = RaylibSupport.raylib370;
} else {
   enum RAYLIB_VERSION_MINOR = 7;
   enum RAYLIB_VERSION_REVISION = 0;
   enum raylibSupport = RaylibSupport.raylib370;
}
